import {Steps, Aside, Tabs, TabItem} from "@astrojs/starlight/components"

<Steps>

1. Create Project.
    ```bash
    npx hu-config@latest # Use this.
    # OR
    npx hucli@latest # In Development.
    ```

2. Follow the Prompts.
    <Aside>
    As of now, use ***hu-config***. Once ***hucli*** is released, ***hu-config*** will be deprecated.
    </Aside>
    <Tabs>
      <TabItem label="hu-config">
      ```text
      - What is your name? (name)
      - Have you used hu-config before? y / n
      - Select the plugins to download: HTMLHint, Marp for VS Code, GitHub Actions
      ```
      </TabItem>
      <TabItem label="hucli (non-custom)">
      ```text
      - Would you like to run the new Nitrox DevKit? Yes
      - Which set would you like to install? Slim / Standard
      - What theme do you want to install? Night / Night Dark
      ```
      </TabItem>
      <TabItem label="hucli (custom)">
      ```text
      - Would you like to run the new Nitrox DevKit? Yes
      - Which set would you like to install? Custom
      - Which Packages would you like to install? HTMLHint, Marp for VS Code, GitHub Actions
      ```
      </TabItem>
    </Tabs>

</Steps>
